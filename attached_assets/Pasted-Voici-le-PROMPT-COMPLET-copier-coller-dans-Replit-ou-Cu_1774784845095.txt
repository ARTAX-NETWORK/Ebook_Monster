#### Classe KeyRotator

```javascript
class QwenKeyRotator {
  constructor(keys) {
    this.keys = keys.map(k => ({ key: k, usage: 0, status: 'active' }));
  }

  getNextKey() {
    const activeKeys = this.keys.filter(k => k.status === 'active');
    const sorted = activeKeys.sort((a, b) => a.usage - b.usage);
    const selected = sorted[0];
    if (selected) {
      selected.usage++;
      return selected.key;
    }
    return null;
  }

  markCooldown(key) {
    const k = this.keys.find(k => k.key === key);
    if (k) {
      k.status = 'cooldown';
      setTimeout(() => { k.status = 'active'; }, 60000);
    }
  }
}
```

```javascript
      source: 'fr',

      format: 'html'
```

## 🧠 API DE GÉNÉRATION (app/api/generate/route.js)

```javascript
import { NextResponse } from 'next/server';
import { generateEbookHTML } from '@/lib/ebook-generator';
import { translateEbookBundle } from '@/lib/translator';
import JSZip from 'jszip';

export async function POST(request) {
  const { 
    ebookTitle, ebookSubtitle, authorName, githubRepoLink, 
    brandName, priceStarter, pricePro, priceLifetime, 
    promoEndDate, apiKeys, selectedLangs 
  } = await request.json();
  
  // 1. Générer le nom de marque si non fourni
  const finalBrandName = brandName || generateBrandName();
  
  // 2. Générer le design unique (couleurs, typo, architecture)
  const design = generateUniqueDesign(finalBrandName);
  
  // 3. Générer le HTML de l'ebook (version française)
  const frenchHTML = generateEbookHTML({
    title: ebookTitle,
    subtitle: ebookSubtitle,
    author: authorName,
    githubLink: githubRepoLink,
    brandName: finalBrandName,
    design,
    prices: { starter: priceStarter, pro: pricePro, lifetime: priceLifetime },
    promoEndDate
  });
  
  // 4. Traduire dans les langues sélectionnées
  const translations = { FR: frenchHTML };
  
  for (const lang of selectedLangs) {
    if (lang === 'FR') continue;
    
    const translated = await translateEbook(frenchHTML, lang, apiKeys);
    translations[lang] = translated;
  }
  
  // 5. Créer le bundle ZIP
  const zip = new JSZip();
  for (const [lang, content] of Object.entries(translations)) {
    zip.file(`ebook-${lang}.html`, content);
  }
  
  const bundle = await zip.generateAsync({ type: 'base64' });
  const bundleId = Date.now().toString();
  
  // Stocker temporairement (en mémoire, pas de DB)
  global.tempBundles = global.tempBundles || {};
  global.tempBundles[bundleId] = bundle;
  
  // Nettoyer après 1 heure
  setTimeout(() => delete global.tempBundles[bundleId], 3600000);
  
  return NextResponse.json({ bundleId, success: true });
}

function generateBrandName() {
  const adjectives = ['Ultimate', 'God', 'Elite', 'Supreme', 'Legendary', 'Infinite', 'Absolute', 'Divine', 'Mighty', 'Prime', 'Apex', 'Zenith', 'Summit', 'Pinnacle', 'Peak'];
  const nouns = ['Factory', 'Engine', 'Machine', 'Forge', 'Lab', 'Studio', 'Workshop', 'Arsenal', 'Hive', 'Nexus', 'Core', 'Matrix', 'Blueprint', 'System', 'Formula'];
  const suffixes = ['Pro', 'AI', 'X', 'Next', 'Ultra', 'Max', 'Elite', 'Prime', 'One', 'Zero', 'Core', 'Plus', 'Labs', 'Tech', 'Flow'];
  
  const adj = adjectives[Math.floor(Math.random() * adjectives.length)];
  const noun = nouns[Math.floor(Math.random() * nouns.length)];
  const suffix = suffixes[Math.floor(Math.random() * suffixes.length)];
  
  return `${adj} ${noun} ${suffix}`;
}

function generateUniqueDesign(seed) {
  const palettes = [
    { primary: '#1A1A2E', secondary: '#E94560', accent: '#0F3460', bg: '#533483' },
    { primary: '#0D1B2A', secondary: '#FFB703', accent: '#1B4332', bg: '#2D6A4F' },

  ];
  
  const index = seed.length % palettes.length;
  return palettes[index];
}
```

```jsx
'use client';
import { useEffect, useState } from 'react';
import { useSearchParams } from 'next/navigation';

export default function GeneratePage() {
  const searchParams = useSearchParams();
  const bundleId = searchParams.get('bundle');
  const [bundleReady, setBundleReady] = useState(false);
  const [languages, setLanguages] = useState([]);
  const [downloading, setDownloading] = useState(false);
  
  useEffect(() => {
    if (bundleId) {
      fetch(`/api/bundle/${bundleId}`)
        .then(res => res.json())
        .then(data => {
          setLanguages(data.languages);
          setBundleReady(true);
        });
    }
  }, [bundleId]);
  
  const downloadBundle = async () => {
    setDownloading(true);
    const response = await fetch(`/api/bundle/${bundleId}/download`);
    const blob = await response.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `ebook-bundle-${bundleId}.zip`;
    a.click();
    URL.revokeObjectURL(url);
    setDownloading(false);
  };
  
  return (
    <div className="min-h-screen bg-gray-50 p-8">
      <div className="max-w-4xl mx-auto text-center">
        {!bundleReady ? (
          <div className="py-20">
            <div className="animate-spin text-4xl mb-4">⏳</div>
            <h2 className="text-xl font-semibold">Génération en cours...</h2>
            <p className="text-gray-500">Création de l'ebook et traductions, veuillez patienter.</p>
          </div>
        ) : (
          <div className="bg-white rounded-lg shadow p-8">
            <div className="text-6xl mb-4">✅</div>
            <h2 className="text-2xl font-bold mb-2">Ebook généré avec succès !</h2>
            <p className="text-gray-600 mb-6">
              {languages.length} versions disponibles
            </p>
            
            <button 
              onClick={downloadBundle}
              disabled={downloading}
              className="bg-green-600 text-white px-8 py-3 rounded-lg font-semibold hover:bg-green-700 transition disabled:opacity-50"
            >
              {downloading ? 'Préparation...' : '⬇️ Télécharger le bundle ZIP'}
            </button>
            
            <div className="mt-8 text-left">
              <h3 className="font-semibold mb-3">Versions incluses :</h3>
              <div className="space-y-2">
                {languages.map(lang => (
                  <div key={lang.code} className="flex items-center gap-3 p-2 bg-gray-50 rounded">
                    <span className="text-2xl">{lang.flag}</span>
                    <span className="font-medium">{lang.name}</span>
                    <span className="text-gray-400 text-sm">({lang.native})</span>
                    <span className="ml-auto text-green-600 text-sm">✓ Prêt</span>
                  </div>
                ))}
              </div>
            </div>
            
            <div className="mt-8 p-4 bg-blue-50 rounded-lg">
              <p className="text-sm text-blue-800">
                💡 <strong>Prochaine étape :</strong> Hébergez ces fichiers sur GitHub Pages ou Netlify, 
                puis vendez l'accès à votre ebook.
              </p>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
```

```javascript
import { NextResponse } from 'next/server';

export async function GET(request, { params }) {
  const { id } = params;
  
  const bundle = global.tempBundles?.[id];
  if (!bundle) {
    return NextResponse.json({ error: 'Bundle not found' }, { status: 404 });
  }
  
  const buffer = Buffer.from(bundle, 'base64');
  
  return new NextResponse(buffer, {
    headers: {
      'Content-Type': 'application/zip',
      'Content-Disposition': `attachment; filename="ebook-bundle-${id}.zip"`,
      'Content-Length': buffer.length.toString()
    }
  });
}
```

## 🔑 SYSTÈME DE ROTATION (lib/qwen-rotator.js)

```javascript
export class QwenKeyRotator {
  constructor(keys) {
    this.keys = keys.map(k => ({ key: k, usage: 0, status: 'active', errors: 0 }));
  }
  
  getNextKey() {
    const activeKeys = this.keys.filter(k => k.status === 'active');
    if (activeKeys.length === 0) return null;
    
    const sorted = activeKeys.sort((a, b) => a.usage - b.usage);
    const selected = sorted[0];
    if (selected) {
      selected.usage++;
      return selected.key;
    }
    return null;
  }
  
  markSuccess(key) {
    const k = this.keys.find(k => k.key === key);
    if (k) {
      k.errors = 0;
    }
  }
  
  markError(key) {
    const k = this.keys.find(k => k.key === key);
    if (k) {
      k.errors++;
      if (k.errors >= 3) {
        k.status = 'cooldown';
        setTimeout(() => {
          k.status = 'active';
          k.errors = 0;
        }, 120000); // 2 minutes de cooldown
      }
    }
  }
  
  resetUsage() {
    this.keys.forEach(k => k.usage = 0);
  }
}

export async function callQwenAPI(apiKey, prompt, rotator) {
  try {
    const response = await fetch('https://dashscope.aliyuncs.com/api/v1/services/aigc/text-generation/generation', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${apiKey}`
      },
      body: JSON.stringify({
        model: 'qwen-plus',
        input: { messages: [{ role: 'user', content: prompt }] },
        parameters: { result_format: 'message' }
      })
    });
    
    if (!response.ok) throw new Error(`API error: ${response.status}`);
    const data = await response.json();
    if (rotator) rotator.markSuccess(apiKey);
    return data.output.choices[0].message.content;
  } catch (error) {
    if (rotator) rotator.markError(apiKey);
    throw error;
  }
}
```
